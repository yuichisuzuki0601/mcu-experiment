# いろいろインストール
brew install python3 pipx
pipx install esptool mpremote

# zshrcにpathを書きこむコマンド
pipx ensurepath

# USBで繋がったesp32のアドレス確認
ls /dev/tty.usb*

# 疎通確認
esptool --port /dev/tty.usbserial-210 read-mac

# 初期化
esptool --port /dev/tty.usbserial-210 erase-flash

# ファームウェアの書き込み
esptool --chip esp32 --port /dev/tty.usbserial-210 --baud 460800 write-flash -z 0x1000 ./device/firmware/ESP32_GENERIC-20260406-v1.28.0.bin
